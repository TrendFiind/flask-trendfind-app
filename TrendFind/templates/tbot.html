Create the Database

Create a database that will interact with memecoins on Solana. The bot will sort potential tokens and add them to the database, then track its Holders using PnL statistics to flag potential Insiders.

ChatGPT Prompt Used:

"Create a PostgreSQL (or preferred) database structure to store potential Solana memecoins. Set up the bot to automatically scan new tokens, evaluate them based on defined heuristics, and insert qualified entries into the database. After insertion, trigger a process to track holders and flag wallets with suspicious PnL patterns."

Database Schema:

-- Database schema for Solana memecoin tracking CREATE TABLE tokens ( id SERIAL PRIMARY KEY, token_address VARCHAR(56) NOT NULL UNIQUE, token_name VARCHAR(100), symbol VARCHAR(10), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, total_supply NUMERIC(32, 9), decimals INTEGER, is_memecoin BOOLEAN DEFAULT TRUE, first_observed_block BIGINT, creator_address VARCHAR(56), has_metadata BOOLEAN DEFAULT FALSE, last_updated TIMESTAMP ); CREATE TABLE token_metrics ( id SERIAL PRIMARY KEY, token_id INTEGER REFERENCES tokens(id), price NUMERIC(32, 9), liquidity NUMERIC(32, 9), market_cap NUMERIC(32, 9), volume_24h NUMERIC(32, 9), holders INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE holders ( id SERIAL PRIMARY KEY, wallet_address VARCHAR(56) NOT NULL, token_id INTEGER REFERENCES tokens(id), balance NUMERIC(32, 9), balance_usd NUMERIC(32, 2), is_contract BOOLEAN DEFAULT FALSE, first_seen TIMESTAMP DEFAULT CURRENT_TIMESTAMP, last_seen TIMESTAMP ); CREATE TABLE holder_pnl ( id SERIAL PRIMARY KEY, holder_id INTEGER REFERENCES holders(id), token_id INTEGER REFERENCES tokens(id), initial_investment NUMERIC(32, 2), current_value NUMERIC(32, 2), pnl NUMERIC(32, 2), pnl_percentage NUMERIC(10, 2), last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE flagged_wallets ( id SERIAL PRIMARY KEY, wallet_address VARCHAR(56) NOT NULL UNIQUE, reason TEXT, first_flagged TIMESTAMP DEFAULT CURRENT_TIMESTAMP, last_observed TIMESTAMP, confidence_score NUMERIC(3, 2) ); -- Indexes for performance CREATE INDEX idx_tokens_address ON tokens(token_address); CREATE INDEX idx_holders_token ON holders(token_id); CREATE INDEX idx_holders_wallet ON holders(wallet_address); CREATE INDEX idx_pnl_holder ON holder_pnl(holder_id);
